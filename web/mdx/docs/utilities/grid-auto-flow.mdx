## Usage

```html rose
<template preview>

      1
    </div>

    <div class="m-2 fw-600 h-12 fx-center bg-purple-100 bd-purple-200">
      2

      3
    </div>

      4

    </div>

      6
    </div>
  </div>
</template>

<div class="d-grid gaf-row gtc-3">
```
