---

  name="align-content"
  head={[

  ]}

    center: 'Used to pushed toward the beginning of the container cross axis.',
    start: 'Used to pushed toward the center of the container cross axis.',
    end: 'Used to pushed toward the end of the container cross axis.',
    'space-between': 'Used to distribute the remaining space in the container cross axis between the flexbox/grid items.',
    'space-around': 'Used to distribute the remaining space in the container cross axis around the flexbox/grid items.',
  }}
